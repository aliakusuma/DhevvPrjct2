    TabCorner.CornerRadius = UDim.new(0, 6)
    TabCorner.Parent = TabButton
    
    -- Hover effect
    TabButton.MouseEnter:Connect(function()
        if currentTab ~= tab.Name then
            TweenService:Create(TabButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(45, 45, 55)}):Play()
        end
    end)
    
    TabButton.MouseLeave:Connect(function()
        if currentTab ~= tab.Name then
            TweenService:Create(TabButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(35, 35, 45)}):Play()
        end
    end)
    
    TabButton.MouseButton1Click:Connect(function()
        currentTab = tab.Name
        updateContent()
        
        -- Update all tab buttons
        for _, btn in pairs(tabButtons) do
            if btn.Name == tab.Name .. "Tab" then
                TweenService:Create(btn, TweenInfo.new(0.3), {BackgroundColor3 = Color3.fromRGB(0, 120, 215)}):Play()
                TweenService:Create(btn, TweenInfo.new(0.3), {TextColor3 = Color3.fromRGB(255, 255, 255)}):Play()
            else
                TweenService:Create(btn, TweenInfo.new(0.3), {BackgroundColor3 = Color3.fromRGB(35, 35, 45)}):Play()
                TweenService:Create(btn, TweenInfo.new(0.3), {TextColor3 = Color3.fromRGB(200, 200, 200)}):Play()
            end
        end
    end)
    
    tabButtons[tab.Name] = TabButton
end

-- Update Content Function
function updateContent()
    -- Clear content
    for _, child in ipairs(ContentContainer:GetChildren()) do
        if child:IsA("Frame") then
            child:Destroy()
        end
    end
    
    -- Get current scripts
    local scripts = scriptsData[currentTab] or {}
    
    -- Create scroll container
    local ScrollingFrame = Instance.new("ScrollingFrame")
    ScrollingFrame.Name = "ScrollingFrame"
    ScrollingFrame.Size = UDim2.new(1, 0, 1, 0)
    ScrollingFrame.BackgroundTransparency = 1
    ScrollingFrame.BorderSizePixel = 0
    ScrollingFrame.ScrollBarThickness = 6
    ScrollingFrame.ScrollBarImageColor3 = Color3.fromRGB(100, 100, 120)
    ScrollingFrame.CanvasSize = UDim2.new(0, 0, 0, #scripts * 50)
    ScrollingFrame.Parent = ContentContainer
    
    -- Create script buttons
    for i, scriptData in ipairs(scripts) do
        local ScriptButton = Instance.new("TextButton")
        ScriptButton.Name = "Script_" .. i
        ScriptButton.Size = UDim2.new(1, -20, 0, 40)
        ScriptButton.Position = UDim2.new(0, 10, 0, 10 + (i-1) * 50)
        ScriptButton.BackgroundColor3 = Color3.fromRGB(40, 40, 50)
        ScriptButton.BorderSizePixel = 0
        ScriptButton.Text = scriptData.Name
        ScriptButton.TextColor3 = Color3.fromRGB(220, 220, 220)
        ScriptButton.TextSize = 14
        ScriptButton.Font = Enum.Font.Gotham
        ScriptButton.TextWrapped = true
        ScriptButton.Parent = ScrollingFrame
        
        local ScriptCorner = Instance.new("UICorner")
        ScriptCorner.CornerRadius = UDim.new(0, 6)
        ScriptCorner.Parent = ScriptButton
        
        -- Execute script
        ScriptButton.MouseButton1Click:Connect(function()
            LoadingText.Text = "Executing: " .. scriptData.Name
            TweenService:Create(LoadingBar, TweenInfo.new(0.5), {Size = UDim2.new(0.5, 0, 1, 0)}):Play()
            
            task.wait(0.5)
            
            -- Execute the script
            local success, errorMsg = pcall(function()
                loadstring(scriptData.Script)()
            end)
            
            if success then
                LoadingText.Text = "‚úì Success: " .. scriptData.Name
                TweenService:Create(LoadingBar, TweenInfo.new(0.3), {BackgroundColor3 = Color3.fromRGB(0, 255, 100)}):Play()
            else
                LoadingText.Text = "‚úó Error: " .. scriptData.Name
                TweenService:Create(LoadingBar, TweenInfo.new(0.3), {BackgroundColor3 = Color3.fromRGB(255, 50, 50)}):Play()
            end
            
            TweenService:Create(LoadingBar, TweenInfo.new(0.5), {Size = UDim2.new(1, 0, 1, 0)}):Play()
            
            task.wait(1)
            
            TweenService:Create(LoadingBar, TweenInfo.new(0.5), {Size = UDim2.new(0, 0, 1, 0)}):Play()
            TweenService:Create(LoadingBar, TweenInfo.new(0.3), {BackgroundColor3 = Color3.fromRGB(0, 170, 255)}):Play()
            LoadingText.Text = "Ready"
        end)
        
        -- Hover effects
        ScriptButton.MouseEnter:Connect(function()
            TweenService:Create(ScriptButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(50, 50, 65)}):Play()
        end)
        
        ScriptButton.MouseLeave:Connect(function()
            TweenService:Create(ScriptButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(40, 40, 50)}):Play()
        end)
    end
    
    -- Update canvas size
    ScrollingFrame.CanvasSize = UDim2.new(0, 0, 0, #scripts * 50)
end

-- Dragging functionality
local dragging = false
local dragInput, dragStart, startPos

local function update(input)
    local delta = input.Position - dragStart
    MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

TopBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

TopBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and input == dragInput then
        update(input)
    end
end)
